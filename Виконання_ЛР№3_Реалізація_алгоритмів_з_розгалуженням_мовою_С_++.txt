    int выбор;

    cin >> выбор;

    switch (выбор) {

        case 2: задание_geom1(); break;
        case 4: задание_if4(); break; // Добавляем ветку для задания If4
        default: cout << "Неверное задание! (Только 1, 2, 4)" << endl;
    }

    system("pause");
    return 0;
}

    // Существующая реализация для задания If1
}

void задание_geom1() {
    // Существующая реализация для задания Geometry
}

void задание_if4() {
    int число1, число2, число3, количество = 0;

    cout << "************* If 4 *****************" << endl;
    cout << "Введите три целых числа: ";
    cin >> число1 >> число2 >> число3;

    if (!cin) {
        cout << "Неверный ввод! Пожалуйста, введите целые числа." << endl;
        return;
    }

    if (число1 > 0) количество++;
    if (число2 > 0) количество++;
    if (число3 > 0) количество++;

    cout << "Количество положительных чисел: " << количество << endl;
}
Теперь, когда вы вводите 4 в качестве номера задачи, программа будет выполнять функцию задание_if4, которая подсчитывает количество положительных чисел во входных данных.
Задание 17:
Вот пример программы на C++, которая решает задачу If17:
#include <iostream>
using namespace std;

void task_if17();

int main() {
    task_if17();

    system("pause");
    return 0;
}

void task_if17() {
    float A, B, C;

    cout << "Введіть три дійсних числа (A, B, C): ";
    cin >> A >> B >> C;

    if ((A < B && B < C) || (A > B && B > C)) {
        // значення впорядковані за зростанням або зменшенням
        A *= 2;
        B *= 2;
        C *= 2;
    } else {
        // значення не впорядковані
        A = -A;
        B = -B;
        C = -C;
    }

    cout << "Нові значення змінних A, B, C: " << A << ", " << B << ", " << C << endl;
}
Программа запрашивает у пользователя три действительных числа A, B, C, затем проверяет, впорядкованы ли они по возрастанию или убыванию. В зависимости от результата программа удваивает значения переменных или заменяет их на противоположные. Новые значения выводятся на экран.
